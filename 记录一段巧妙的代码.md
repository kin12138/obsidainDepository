# 统计字符在字符串中第一次出现的位置，并返回它在字符串中出现的总次数。
```c++
#include <string>
#include <iostream>

std::string::size_type find_char(const std::string &s, char c, std::string::size_type &occurs) 
{

	auto ret = s.size();
	occurs = 0;//总次数
	
	for (decltype(ret) i = 0; i != s.size(); ++i)
	{
		if (s[i] == c)
		{
			if (ret == s.size())
			{
				ret = i;//第一次出现位置
			}
		++occurs;
		}

	}
	return ret;
}

int main()
{

	std::string s = "Iamfsdafgsagfdga";
	std::string::size_type occours = 0;//总次数
	auto count = find_char(s, 'a', occours);//第一次出现位置
	std::cout << occours << "\n";
	std::cout << count << std::endl;
	return 0;
}
```
